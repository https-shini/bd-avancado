create table tbCategoria(
    idCategoria number primary key,
    Categoria varchar2(40)
);

create table tbStatusNoticia(
    idStatusNoticia number primary key,
    status varchar2(40)
);

create table tbNoticia(
    idNoticia number primary key,
    titulo varchar2(40),
    conteudo varchar2(40),
    dataCriacao date,
    idStatusNoticia number,
    idCategoria number,
    foreign key (idStatusNoticia) references tbStatusNoticia(idStatusNoticia),
    foreign key (idCategoria) references tbCategoria (idCategoria)
);

create table tbJornalista(
    idJornalista number primary key,
    nome varchar2(40),
    email varchar2(40),
    telefone varchar2(40),
    dataContratacao date
);

create table tbNoticiaJornalista(
    idNoticiaJornalista number primary key,
    idNoticia number,
    idJornalista number,
    foreign key (idNoticia) references tbNoticia(idNoticia),
    foreign key (idJornalista) references tbJornalista (idJornalista)
);
